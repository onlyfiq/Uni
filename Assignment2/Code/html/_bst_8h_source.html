308{
309 if (node == nullptr)
310 {
311 return nullptr;
312 }
313
314 if (value < node->data)
315 {
316 node->left = deleteNode(node->left, value);
317 }
318 else if (value > node->data)
319 {
320 node->right = deleteNode(node->right, value);
321 }
322 else
323 {
324 // Case 1: Leaf
325 if (node->left == nullptr && node->right == nullptr)
326 {
327 delete node;
328 node = nullptr;
329 }
330 // Case 2: One child
331 else if (node->left == nullptr)
332 {
333 Node* temp = node->right;
334 delete node;
335 node = temp;
336 }
337 else if (node->right == nullptr)
338 {
339 Node* temp = node->left;
340 delete node;
341 node = temp;
342 }
343 // Case 3: Two children
344 else
345 {
346 Node* successor = node->right;
347 while (successor->left != nullptr)
348 {
349 successor = successor->left;
350 }
351 node->data = successor->data;
352 node->right = deleteNode(node->right, successor->data);
353 }
354 }
355 return node;
